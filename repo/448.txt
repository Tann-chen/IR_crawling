   var previousSearchTerm='';
   function setPreviousSearchTerm()
     {
	   var flag = false;
  	   if(document.getElementById('searchString').value.trim()!='' && previousSearchTerm.trim() == document.getElementById('searchString').value.trim())
  		   {
  		   flag = false;
  		   }
  	   else
  		   {
  		   flag = true;
  		   }
  	
  	 previousSearchTerm=document.getElementById('searchString').value.trim();
  	 return flag;
     }
